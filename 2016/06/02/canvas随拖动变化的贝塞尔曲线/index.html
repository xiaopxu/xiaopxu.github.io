canvas随拖动变化的贝塞尔曲线

canvas实现贝塞尔曲线模拟

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>贝塞尔曲线</title>
    <style>
        *{
            margin: 0;
            padding: 0;
        }
        #canvas{
            margin: 200px auto;
            display: block;
            border: 1px solid #000;
        }
        #cover{
            width: 500px;
            height: 500px;
            margin: 200px auto;
            position: relative;
            left: 0;
            top: -700px;
        }
        #left{
            width: 26px;
            height: 26px;
            background-color: skyblue;
            border-radius: 50%;
            position: absolute;
            left: 50px;
            top: 50px;
        }
        #right{
            width: 26px;
            height: 26px;
            background-color: pink;
            border-radius: 50%;
            position: absolute;
            left: 450px;
            top: 450px;
        }
        .active{
            box-shadow: 0 0 10px yellow;
        }
        .start{
            width: 24px;
            height: 24px;
            border: 1px solid #ccc;
            border-radius: 50%;
            position: absolute;
            left: -13px;
            bottom: -13px;
            background-color: #fff;
        }
        .end{
            width: 24px;
            height: 24px;
            border: 1px solid #ccc;
            border-radius: 50%;
            position: absolute;
            right: -13px;
            top: -13px;
            background-color: #fff;
        }
    </style>
</head>
<body>
    <canvas id="canvas" width="500" height="500"></canvas>
    <div id="cover">
        <div id="left"></div>
        <div id="right"></div>
        <div class="start"></div>
        <div class="end"></div>
    </div>
</body>
<script>

    var canvas = document.getElementById('canvas');
    var context =  canvas.getContext('2d');

    var btn1 = document.getElementById('left');
    var btn2 = document.getElementById('right');

    //画布宽高
    var w = canvas.width;
    var h = canvas.height;

    //左右拖动点的中心点位置
    var L1 = btn1.offsetLeft + btn1.offsetWidth / 2;
    var T1 = btn1.offsetTop + btn1.offsetHeight / 2;

    var L2 = btn2.offsetLeft + btn2.offsetWidth / 2;
    var T2 = btn2.offsetTop + btn2.offsetHeight / 2;

    //绘制背景上的浅灰色长条
    function bacImg () {
        context.beginPath();
        for (var i = 0; i < 10; i++) {
            context.fillStyle = "#eee";
            context.rect(0,50 * i,w,25);
            context.fill();
        }
    }

    //绘制拖动点到边角的连线
    function p2p () {
        context.beginPath();
        context.strokeStyle = "555";
        context.lineWidth = 1;

        context.moveTo(L1,T1);
        context.lineTo(w,0);
        context.stroke();

        context.moveTo(L2,T2);
        context.lineTo(0,h);
        context.stroke();
    }

    //绘制贝塞尔曲线
    function bezier () {
        context.beginPath();
        context.strokeStyle = "#000";
        context.lineWidth = 5;

        context.moveTo(0,h)
        context.bezierCurveTo(L1,T1,L2,T2,w,0)
        context.stroke(); 
    }

    bacImg();
    p2p();
    bezier();

    //左拖动点事件
    btn1.onmousedown = function (e) {

        var ev = e||window.event;

        var x = ev.clientX - canvas.offsetLeft;
        var y = ev.clientY - canvas.offsetTop;

        var offsetx = x - btn1.offsetLeft;
        var offsety = y - btn1.offsetTop;

        btn1.className = "active"

        document.onmousemove = function (e) {

            var ev = e||window.event;

            var x = ev.clientX - canvas.offsetLeft;
            var y = ev.clientY - canvas.offsetTop;

            btn1.style.left = x - offsetx + "px";
            btn1.style.top = y - offsety + "px";

            L1 = btn1.offsetLeft + btn1.offsetWidth / 2;
            T1 = btn1.offsetTop + btn1.offsetHeight / 2;

            context.clearRect(0,0,w,h);
            bacImg();
            p2p();
            bezier();
        }
        btn1.onmouseup = function () {
            document.onmousemove = null;
            btn1.className = "";
        }
    }  

    //右拖动点事件
    btn2.onmousedown = function (e) {

        var ev = e||window.event;

        var x = ev.clientX - canvas.offsetLeft;
        var y = ev.clientY - canvas.offsetTop;

        var offsetx = x - btn2.offsetLeft;
        var offsety = y - btn2.offsetTop;

        btn2.className = "active"

        document.onmousemove = function (e) {

            var ev = e||window.event;

            var x = ev.clientX - canvas.offsetLeft;
            var y = ev.clientY - canvas.offsetTop;

            btn2.style.left = x - offsetx + "px";
            btn2.style.top = y - offsety + "px";

            L2 = btn2.offsetLeft + btn2.offsetWidth / 2;
            T2 = btn2.offsetTop + btn2.offsetHeight / 2;

            context.clearRect(0,0,w,h);
            bacImg();
            p2p();
            bezier();
        }

        btn2.onmouseup = function () {
            document.onmousemove = null;
            btn2.className = "";
        }
    }

</script>

</html>
